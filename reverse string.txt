
// java program to reverse the letters present in the given string

public class reverse_string {

    public static void main(String args[]){

        String s = "Hello World";

        String rev = " ";

        for(int i = s.length()-1; i>=0; i--)
        {
            rev = rev + s.charAt(i);
        }
        
        System.out.println(rev);
    }
    
}
